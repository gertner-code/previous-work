CREATE VIEW Cost AS
SELECT
  rentalId,
  customerId,
  (
    price *(
      DATEDIFF(startdate,
      enddate) +1
    ) +(tank * 5) + damage
  ) AS totalCost
FROM
  rental r
JOIN
  vehicles v ON v.vehicleId = r.vehicleId
JOIN
  price p ON p.size = v.size;
